# 第2章 数据模型

## 2.1 数据模型层次结构

```
BookChapter (章节元数据)
  └── TextChapter (章节排版结果)
        └── TextPage (页面)
              └── TextLine (行)
                    └── BaseColumn (列/字符)
                          ├── TextColumn (普通文字)
                          ├── TextHtmlColumn (HTML文字)
                          ├── ImageColumn (图片)
                          ├── ReviewColumn (段评)
                          └── ButtonColumn (按钮)
```

## 2.2 BookChapter — 章节元数据

```kotlin
data class BookChapter(
    var url: String = "",               // 章节地址
    var title: String = "",             // 章节标题
    var isVolume: Boolean = false,      // 是否是卷名
    var baseUrl: String = "",           // 用来拼接相对url
    var bookUrl: String = "",           // 书籍地址
    var index: Int = 0,                 // 章节序号
    var isVip: Boolean = false,         // 是否VIP
    var isPay: Boolean = false,         // 是否已购买
    var resourceUrl: String? = null,    // 音频真实URL
    var tag: String? = null,            // 更新时间或其他附加信息
    var wordCount: String? = null,      // 本章节字数
    var start: Long? = null,            // 章节起始位置(本地书)
    var end: Long? = null,              // 章节终止位置(本地书)
    var startFragmentId: String? = null, // EPUB fragmentId
    var endFragmentId: String? = null,   // EPUB fragmentId
    var variable: String? = null,       // 变量
    var imgUrl: String? = null          // 标题段评图或视频封面
)
```

## 2.3 TextChapter — 章节排版结果

```kotlin
data class TextChapter(
    val chapter: BookChapter,           // 原始章节元数据
    val position: Int,                  // 章节在书籍中的位置
    val title: String,                  // 显示标题
    val chaptersSize: Int,              // 总章节数
    val sameTitleRemoved: Boolean,      // 是否去除了重复标题
    val isVip: Boolean,
    val isPay: Boolean,
    val effectiveReplaceRules: List<ReplaceRule>?
) : LayoutProgressListener {

    private val textPages = arrayListOf<TextPage>()   // 页面列表
    private var layout: TextChapterLayout? = null     // 排版任务

    val pages: List<TextPage>           // 页面列表(只读)
    val layoutChannel: Channel<TextPage> // 排版进度通道
    val pageSize: Int                   // 总页数
    var isCompleted: Boolean            // 排版是否完成

    fun getPage(index: Int): TextPage?                    // 获取指定页
    fun getPageByReadPos(readPos: Int): TextPage?         // 通过字符位置获取页
    fun getPageIndexByCharIndex(charIndex: Int): Int       // 字符位置→页码(二分查找)
    fun createLayout(scope, book, bookContent)             // 创建排版任务
}
```

## 2.4 TextPage — 页面数据模型

```kotlin
data class TextPage(
    var index: Int = 0,                          // 页面索引
    var text: String = "正在加载中...",              // 页面文本
    var title: String = "正在加载中...",             // 章节标题
    private val textLines: ArrayList<TextLine>,   // 行列表
    var chapterSize: Int = 0,                     // 总章节数
    var chapterIndex: Int = 0,                    // 章节索引
    var height: Float = 0f,                       // 页面总高度
    var leftLineSize: Int = 0,                    // 左列行数(双页模式)
    var renderHeight: Int = 0                     // 渲染高度
) {
    val lines: List<TextLine>                     // 行列表(只读)
    val lineSize: Int                             // 行数
    val charSize: Int                             // 字符数
    val chapterPosition: Int                      // 该页在章节中的起始字符位置
    val searchResult = hashSetOf<TextBaseColumn>() // 搜索结果高亮列集合
    var isMsgPage: Boolean = false                // 是否消息页
    var canvasRecorder = CanvasRecorderFactory.create(true)  // Canvas录制器
    var doublePage = false                        // 是否双页模式
    var isCompleted = Boolean                     // 该页排版是否完成

    fun draw(view, canvas, relativeOffset)        // 绘制
    fun upLinesPosition()                         // 底部对齐
    fun containPos(chapterPos: Int): Boolean       // 判断位置是否在当前页
}
```

## 2.5 TextLine — 行数据模型

```kotlin
data class TextLine(
    var text: String = "",                        // 行文本
    private val textColumns: ArrayList<BaseColumn>, // 列列表
    var lineTop: Float = 0f,                      // 行顶部Y坐标
    var lineBase: Float = 0f,                     // 行基线Y坐标
    var lineBottom: Float = 0f,                   // 行底部Y坐标
    var indentWidth: Float = 0f,                  // 缩进宽度
    var paragraphNum: Int = 0,                    // 段落编号
    var chapterPosition: Int = 0,                 // 该行在章节中的起始字符位置
    val isTitle: Boolean = false,                 // 是否标题行
    var isParagraphEnd: Boolean = false,          // 是否段落结束
    var isImage: Boolean = false,                 // 是否图片行
    var startX: Float = 0f,                       // 行起始X坐标
    var indentSize: Int = 0,                      // 缩进字符数
    var extraLetterSpacing: Float = 0f,           // 额外字间距(两端对齐用)
    var extraLetterSpacingOffsetX: Float = 0f,    // 字间距偏移X
    var wordSpacing: Float = 0f,                  // 词间距(两端对齐用)
    var exceed: Boolean = false                   // 是否超出可视区域
) {
    val columns: List<BaseColumn>                 // 列列表(只读)
    val height: Float                             // 行高度
    var canvasRecorder = CanvasRecorderFactory.create()  // Canvas录制器

    fun upTopBottom(durY, textHeight, fontMetrics) // 更新行高
    fun isTouch(x, y, relativeOffset): Boolean     // 触摸检测
    fun draw(view, canvas)                         // 绘制
}
```

## 2.7 Column 列类型体系

### 接口层次

```
BaseColumn (interface)
│   var start: Float        // 列起始X坐标
│   var end: Float          // 列结束X坐标
│   var textLine: TextLine  // 所属行
│   fun draw(canvas)        // 绘制
│   fun isTouch(x): Boolean // 触摸检测
│
├── TextBaseColumn (interface, extends BaseColumn)
│   │   val charData: String      // 字符数据
│   │   var selected: Boolean     // 是否被选中
│   │   var isSearchResult: Boolean // 是否搜索结果
│   │
│   ├── TextColumn (data class)        // 普通文字列
│   └── TextHtmlColumn (data class)    // HTML文字列(带颜色/大小/链接)
│
├── ImageColumn (data class)           // 图片列
│       var src: String           // 图片地址
│       var click: String?        // 点击动作
│
├── ReviewColumn (data class)          // 段评列
│       val count: Int            // 评论数
│
└── ButtonColumn (data class)          // 按钮列(空实现)
```

# 第4章 视图层

## 4.1 ReadView — 阅读视图容器

继承 `FrameLayout`，实现 `DataSource` 和 `LayoutProgressListener` 接口。

### 核心字段

```kotlin
class ReadView(context, attrs) : FrameLayout(context, attrs), DataSource, LayoutProgressListener {

    var pageFactory: TextPageFactory = TextPageFactory(this)
    var pageDelegate: PageDelegate? = null        // 翻页动画委托

    val prevPage by lazy { PageView(context) }    // 上一页
    val curPage by lazy { PageView(context) }     // 当前页
    val nextPage by lazy { PageView(context) }    // 下一页

    val autoPager = AutoPager(this)               // 自动翻页器

    // 触摸坐标
    var startX, startY, lastX, lastY, touchX, touchY: Float

    // 九宫格触摸区域
    private val tlRect, tcRect, trRect = RectF()  // 上：左/中/右
    private val mlRect, mcRect, mrRect = RectF()  // 中：左/中/右
    private val blRect, bcRect, brRect = RectF()  // 下：左/中/右
}
```

### 子视图添加顺序

```kotlin
init {
    addView(nextPage)   // 先添加(最底层)
    addView(curPage)
```

### upPageAnim — 切换翻页动画

```kotlin
fun upPageAnim() {
    when (ReadBook.pageAnim()) {
        PageAnim.coverPageAnim -> pageDelegate = CoverPageDelegate(this)
        PageAnim.slidePageAnim -> pageDelegate = SlidePageDelegate(this)
        PageAnim.simulationPageAnim -> pageDelegate = SimulationPageDelegate(this)
        PageAnim.scrollPageAnim -> pageDelegate = ScrollPageDelegate(this)
        else -> pageDelegate = NoAnimPageDelegate(this)
    }
}
```

### onTouchEvent — 触摸事件处理

```kotlin
override fun onTouchEvent(event: MotionEvent): Boolean {
    when (event.action) {
        ACTION_DOWN -> {
            if (isTextSelected) { curPage.cancelSelect() }
            postDelayed(longPressRunnable, 600L)   // 长按检测
            pageDelegate?.onTouch(event)
            pageDelegate?.onDown()
            setStartPoint(event.x, event.y)
        }
        ACTION_MOVE -> {
            if (!isMove) isMove = distance > slopSquare
            if (isMove) {
                if (isTextSelected) selectText(event.x, event.y)
                else pageDelegate?.onTouch(event)
            }
        }
        ACTION_UP -> {
            if (!isMove && !longPressed) {
                if (!curPage.onClick(startX, startY)) onSingleTapUp()
            }
            pageDelegate?.onTouch(event)
        }
    }
    return true
}
```

```kotlin
private fun onSingleTapUp() {
    when {
        mcRect.contains(startX, startY) -> click(AppConfig.clickActionMC)  // 正中
        bcRect.contains(startX, startY) -> click(AppConfig.clickActionBC)  // 中下
        // ... 其他7个区域
    }
}

private fun click(action: Int) {
    when (action) {
        0  -> callBack.showActionMenu()        // 显示菜单
        1  -> pageDelegate?.nextPageByAnim(100) // 下一页
        2  -> pageDelegate?.prevPageByAnim(100) // 上一页
        3  -> ReadBook.moveToNextChapter(true)  // 下一章
        4  -> ReadBook.moveToPrevChapter(true)  // 上一章
        // ... 其他动作
    }
}
```

## 4.3 ContentTextView — 文本绘制视图

继承 `View`，是文本内容 **实际渲染** 的地方。

### 核心字段

```kotlin
class ContentTextView(context, attrs) : View(context, attrs) {
    var textPage: TextPage = TextPage()
    private var pageOffset = 0                    // 滚动偏移(滚动模式)
    private var autoPager: AutoPager? = null

    companion object {
        private val renderThread = Executors.newSingleThreadExecutor()
    }
}
```

### onDraw — 绘制入口

```kotlin
override fun onDraw(canvas: Canvas) {
    canvas.clipRect(visibleRect)
    drawPage(canvas)
}

private fun drawPage(canvas: Canvas) {
    var relativeOffset = relativeOffset(0)
    textPage.draw(this, canvas, relativeOffset)   // 绘制当前页

    if (!callBack.isScroll) return

    // 滚动模式: 绘制后续页面
    if (pageFactory.hasNext()) {
        val textPage1 = relativePage(1)
        relativeOffset += textPage.height
        textPage1.draw(this, canvas, relativeOffset)
    }
}
```

### scroll — 滚动事件处理

```kotlin
fun scroll(mOffset: Int) {
    pageOffset += mOffset
    // 边界检查 + 自动切页
    if (pageOffset > 0) {
        if (pageFactory.moveToPrev(true)) pageOffset -= textPage.height.toInt()
    } else if (pageOffset < -textPage.height) {
        if (pageFactory.moveToNext(true)) pageOffset += textPage.height.toInt()
    }
    postInvalidate()
}
```

### preRenderPage — 预渲染

```kotlin
private fun preRenderPage() {
    renderThread.execute {
        pageFactory.run {
            if (hasPrev()) prevPage.render(this@ContentTextView)
            curPage.render(this@ContentTextView)
            if (hasNext()) nextPage.render(this@ContentTextView)
        }
        postInvalidate()
    }
}
```

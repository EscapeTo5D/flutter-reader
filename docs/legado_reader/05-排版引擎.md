# 第5章 排版引擎

## 5.1 排版引擎架构

```
ContentProcessor.getContent()
  → BookContent (段落列表)
    → TextChapterLayout (排版实现)
      → setTypeText() / setTypeImage() / setTypeHtml()
        → StaticLayout 或 ZhLayout 计算行布局
        → 逐字符生成 Column
        → prepareNextPageIfNeed() 分页
        → Channel<TextPage> 输出页面
```

## 5.2 ChapterProvider — 排版配置单例

全局单例对象，持有所有排版相关的全局状态。

### 全部字段

```kotlin
object ChapterProvider {
    // 视图尺寸
    var viewWidth = 0
    var viewHeight = 0

    // 内边距
    var paddingLeft = 0;  var paddingTop = 0
    var paddingRight = 0; var paddingBottom = 0

    // 可绘制区域
    var visibleWidth = 0       // 可绘制宽度
    var visibleHeight = 0      // 可绘制高度
    var visibleRight = 0
    var visibleBottom = 0

    // 排版参数
    var lineSpacingExtra = 0f      // 行间距(乘数)
    var paragraphSpacing = 0       // 段落间距
    var titleTopSpacing = 0
    var titleBottomSpacing = 0
    var indentCharWidth = 0f       // 缩进字符宽度

    // 画笔
    var titlePaint: TextPaint = TextPaint()      // 标题画笔
    var contentPaint: TextPaint = TextPaint()    // 内容画笔

    // 双页模式
    var doublePage = false
}
```

### upStyle — 从配置更新样式

```kotlin
fun upStyle() {
    typeface = getTypeface(ReadBookConfig.textFont)
    getPaints(typeface).let { titlePaint = it.first; contentPaint = it.second }
    lineSpacingExtra = ReadBookConfig.lineSpacingExtra / 10f
    paragraphSpacing = ReadBookConfig.paragraphSpacing
    indentCharWidth = StaticLayout.getDesiredWidth(bodyIndent, contentPaint) / bodyIndent.length
    upLayout()
}
```

### upLayout — 计算布局参数

```kotlin
fun upLayout() {
    // 双页模式判断
    doublePage = when (AppConfig.doublePageHorizontal) {
        "0" -> false
        "1" -> true
        "2" -> (viewWidth > viewHeight) && ReadBook.pageAnim() != 3
        "3" -> (viewWidth > viewHeight || appCtx.isPad) && ReadBook.pageAnim() != 3
    }

    // 计算可绘制区域
    visibleWidth = if (doublePage) viewWidth / 2 - paddingLeft - paddingRight
                   else viewWidth - paddingLeft - paddingRight
    visibleHeight = viewHeight - paddingTop - paddingBottom
}
```

## 5.3 TextChapterLayout — 章节排版实现

### 排版主流程

```kotlin
private suspend fun getTextChapter(...) {
    // 1. 处理标题
    if (titleMode != 2) setTypeText(book, displayTitle, titlePaint, ..., isTitle = true)

    // 2. 遍历 contents 列表
    for (contentList in contents) {
        for (content in contentList) {
            when {
                content.startsWith("<img") -> setTypeImage(...)
                content.startsWith("<") -> setTypeHtml(...)
                else -> setTypeText(...)
            }
        }
    }

    // 3. 完成最后一页
    if (pendingTextPage.lines.isNotEmpty()) onPageCompleted()
    channel.close()
}
```

### setTypeText — 文字排版核心

```kotlin
private suspend fun setTypeText(book, text, textPaint, textHeight, ...) {
    // 1. 创建 Layout
    val layout = if (useZhLayout) ZhLayout(text, textPaint, visibleWidth, ...)
                 else StaticLayout(text, textPaint, visibleWidth, ...)

    // 2. 遍历每一行
    for (lineIndex in 0 until layout.lineCount) {
        prepareNextPageIfNeed(durY + textHeight)   // 检查分页
        val lineText = text.substring(lineStart, lineEnd)

        // 3. 根据行位置选择排列策略
        when {
            isFirstLine && isMultiLine -> addCharsToLineFirst(...)  // 首行: 缩进+两端对齐
            isLastLine -> addCharsToLineNatural(...)                // 末行: 自然排列
            else -> addCharsToLineMiddle(...)                       // 中间行: 两端对齐
        }

        textLine.upTopBottom(durY, textHeight, fontMetrics)
        durY += textHeight * lineSpacingExtra
    }
    durY += textHeight * paragraphSpacing / 10f
}
```

### 三种字符排列策略

#### addCharsToLineFirst — 首行(有缩进+两端对齐)

```kotlin
// 1. 绘制缩进字符
repeat(bodyIndent.length) {
    textLine.addColumn(TextColumn(charData = indentChar, start = x, end = x + indentCharWidth))
    x += indentCharWidth
}
// 2. 剩余字符两端对齐
addCharsToLineMiddle(..., startX = x)
```

#### addCharsToLineMiddle — 中间行(无缩进+两端对齐)

```kotlin
val residualWidth = visibleWidth - desiredWidth
val spaceCount = words.count { it == " " }

if (spaceCount > 1) {
    // 策略A: 有空格 → 分配到空格上
    val d = residualWidth / spaceCount
    textLine.wordSpacing = d
} else {
    // 策略B: 无空格(中文) → 分配到字符间隙上
    val gapCount = words.lastIndex
    val d = residualWidth / gapCount
    textLine.extraLetterSpacing = d / textPaint.textSize
}
```

#### addCharsToLineNatural — 末行/标题(自然排列)

```kotlin
// 按自然宽度排列，无额外间距
for (index in words.indices) {
    val x1 = x + charWidth
    addCharToLine(book, textLine, char, x, x1)
    x = x1
}
```

### prepareNextPageIfNeed — 分页逻辑

```kotlin
private suspend fun prepareNextPageIfNeed(requestHeight: Float = -1f) {
    if (requestHeight > visibleHeight || requestHeight == -1f) {
        if (doublePage && absStartX < viewWidth / 2) {
            // 双页模式: 左列结束, 切换到右列
            absStartX = viewWidth / 2 + paddingLeft
        } else {
            // 当前页结束
            onPageCompleted()
            pendingTextPage = TextPage()
            absStartX = paddingLeft
        }
        durY = 0f
    }
}
```

## 5.4 TextPageFactory — 页面工厂

```kotlin
class TextPageFactory(dataSource: DataSource) : PageFactory<TextPage>(dataSource) {

    override val curPage: TextPage get() = currentChapter?.getPage(pageIndex)

    override val nextPage: TextPage get() {
        currentChapter?.let {
            if (pageIndex < it.pageSize - 1) return it.getPage(pageIndex + 1)
        }
        return nextChapter?.getPage(0)
    }

    override fun moveToNext(upContent: Boolean): Boolean {
        if (currentChapter?.isLastIndex(pageIndex) == true) {
            ReadBook.moveToNextChapter(upContent, false)  // 跨章
        } else {
            ReadBook.setPageIndex(pageIndex + 1)          // 章内
        }
    }

    override fun moveToPrev(upContent: Boolean): Boolean {
        if (pageIndex <= 0) {
            ReadBook.moveToPrevChapter(upContent)         // 跨章
        } else {
            ReadBook.setPageIndex(pageIndex - 1)          // 章内
        }
    }
}
```

## 5.5 ZhLayout — 中文排版布局

解决 Android 原生 StaticLayout 不支持标点压缩的问题。

### 标点分类

```kotlin
// 禁止出现在行首的标点(行尾标点)
private val postPanc = hashSetOf("，","。","：","？","！","、", ...)

// 禁止出现在行尾的标点(行首标点)
private val prePanc = hashSetOf(""","（","《","【", ...)
```

### 断行模式

```kotlin
enum class BreakMod {
    NORMAL,         // 正常断行
    BREAK_ONE_CHAR, // 下移一个字(解决标点冲突)
    BREAK_MORE_CHAR,// 下移多个字(连续标点)
    CPS_1,          // 两个后置标点压缩(不换行)
    CPS_2,          // 前置+前置+字 压缩
    CPS_3           // 前置+字+后置 压缩
}
```

### 标点压缩偏移

```kotlin
// 后置标点左边缘压缩
private fun getPostPancOffset(string: String): Float {
    return max(textRect.left - gap, 0f)
}

// 前置标点右边缘压缩
private fun getPrePancOffset(string: String): Float {
    val d = max(cnCharWidth - textRect.right - gap, 0f)
    return cnCharWidth / 2 - d
}
```
